inline MATH.DIVCHECK		// FFF
{
    DATAF tmpf:#
	DATA8 flag:#
	DIVF :0 :1 tmpf:#
	CP_EQF 0.0 :1 flag:#
	SELECTF flag:# 0.0 tmpf:# :2
}

{
	IN_F range
	OUT_F result
	
	DATA16 range_16
	DATA16 value
	MOVEF_16 range range_16
	
	RANDOM 1 range_16 value
	MOVE16_F value result
}
